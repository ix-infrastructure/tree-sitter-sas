==================
Macro definition no parameters
==================

%macro setup;
  %let env = prod;
%mend setup;

---

(program
  (macro_definition
    name: (macro_name
      (identifier))
    (macro_variable_assignment
      name: (identifier))
    (macro_end
      (macro_name
        (identifier)))))
